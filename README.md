  ( how fun1(),fun2(),fun3() differs from each other)
    
 ``` 
 #include<bits/stdc++.h>
 using namespace std;
 void fun1(int &r){ r++;}
 void fun2(int *p1){ (*p1)++;}
 void fun3(int *p2){	*p2++; }
 int  main(){
	int a=5,*p=&a; 
	fun1(a);
	cout<<a<<endl;
	fun2(*&p);
	cout<<a<<endl;
	fun3(&(*&a));
	cout<<a<<endl;
	return 0;
 }
 ``` 
